        function renderTabs(prefix, data, titleKey) {
            const tabsHeader = document.getElementById(`${prefix}-tabs`);
            const tabsContent = document.getElementById(`${prefix}-content`);
            
            data.forEach((item, index) => {
                const btn = document.createElement('button');
                btn.className = `tab-btn ${index === 0 ? 'active' : ''}`;
                btn.innerText = item.tab || item[titleKey];
                btn.onclick = () => switchTab(prefix, index);
                tabsHeader.appendChild(btn);

                const pane = document.createElement('div');
                pane.className = `tab-pane ${index === 0 ? 'active' : ''}`;
                pane.id = `${prefix}-pane-${index}`;
                pane.innerHTML = `
                    <div class="card">
                        <div class="card-header">
                            <div>
                                <div class="card-title">${item[titleKey]}</div>
                                <div style="font-weight: 600; font-size: 1rem;">${item.role || ''}</div>
                            </div>
                            <div class="card-meta">${item.period}</div>
                        </div>
                        <ul>${item.points.map(p => `<li>${p}</li>`).join('')}</ul>
                    </div>
                `;
                tabsContent.appendChild(pane);
            });
        }

        function switchTab(prefix, index) {
            document.querySelectorAll(`#${prefix}-tabs .tab-btn`).forEach((b, i) => {
                b.classList.toggle('active', i === index);
            });
            document.querySelectorAll(`#${prefix}-content .tab-pane`).forEach((p, i) => {
                p.classList.toggle('active', i === index);
            });
        }
